def eval_postfix(tokens: list[str]) -> int:
    # Use a stack
    # When you see a number: push it
    # When you see an operator: pop right operand, pop left operand, apply and push result
    
    # Empty stack to store numbers
    stack = []

    # Loop through the numbers
    for i in tokens:
        
        # I attempted to use isinstance, but it didn't work because whenever it check "3" it would fail. So I use isdigit to check it
        if i.isdigit():
            # Appends the index, which is converted into an integer
            stack.append(int(i))
        else:
            # With the instructions, the right is removed and the left is removed
            right = stack.pop()
            left = stack.pop()
            

            # If statements to check what operation needs to be done
            if i == "+":
                # Depending on the operation, the result is found and added to the stack
                stack.append(left + right)
            elif i == "-":
                stack.append(left - right)
            elif i == "*":
                stack.append(left * right)
            elif i == "/":
                stack.append(left // right)
    return stack[0]
    
print(eval_postfix("3 4 +".split())) 
print(eval_postfix("10 2 3 * +".split())) 
print(eval_postfix("8 3 - 2 *".split()))
